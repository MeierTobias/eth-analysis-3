% colors
\usepackage{color,xcolor}
\definecolor{section}{HTML}{1c59c9}
\definecolor{subsection}{HTML}{568ae8}
\definecolor{subsubsection}{HTML}{87a9e8}
\definecolor{titletext}{RGB}{255,255,255}

% colored box for section
\setkomafont{section}{\mysection}
\newcommand{\mysection}[1]{%
    \setlength{\fboxsep}{0cm}%already boxed
    \colorbox{section}{%
        \begin{minipage}{\linewidth}%
            \vspace*{2pt}%Space before
            \hspace{2pt}%left indent
            \textcolor{titletext}{#1}
            \vspace*{2pt}%Space after
        \end{minipage}%
    }}
% colored box for subsection
\setkomafont{subsection}{\mysubsection}
\newcommand{\mysubsection}[1]{%

    \colorbox{subsection}{%

            \vspace*{2pt}%Space after

    }}

  afterindent=false,

  afterskip=.25\baselineskip]{section}
\RedeclareSectionCommand[
  %runin=false,
  afterindent=false,
  beforeskip=.25\baselineskip,
  afterskip=.25\baselineskip]{subsection}
